\item bitsets (see \hyperlink{classlgraph_1_1utils_1_1static__bitset}{static\+\_\+bitset})
\item wrappers for random number generators (see \hyperlink{classlgraph_1_1utils_1_1random__generator}{random\+\_\+generator})
\end{DoxyItemize}
